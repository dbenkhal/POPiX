01492         internal static Expr Create(Context ctx, FuncDecl f, params Expr[] arguments)
01493         {
01494             Contract.Requires(ctx != null);
01495             Contract.Requires(f != null);

01497 
01498             IntPtr obj = Native.Z3_mk_app(ctx.nCtx, f.NativeObject, 
01499                                           AST.ArrayLength(arguments), 
01500                                           AST.ArrayToNative(arguments));
01501             return Create(ctx, obj);

01503 
01504         [Pure]
01505         new internal static Expr Create(Context ctx, IntPtr obj)
01506         {
01507             Contract.Requires(ctx != null);
01508             Contract.Ensures(Contract.Result<Expr>() != null);
01509 
01510             Z3_ast_kind k = (Z3_ast_kind)Native.Z3_get_ast_kind(ctx.nCtx, obj);
01511             if (k == Z3_ast_kind.Z3_QUANTIFIER_AST)
01512                 return new Quantifier(ctx, obj);
01513             IntPtr s = Native.Z3_get_sort(ctx.nCtx, obj);
01514             Z3_sort_kind sk = (Z3_sort_kind)Native.Z3_get_sort_kind(ctx.nCtx, s);
01515 
01516             if (Native.Z3_is_algebraic_number(ctx.nCtx, obj) != 0) // is this a numeral ast?
01517                 return new AlgebraicNum(ctx, obj);
01518 
01519             if (Native.Z3_is_numeral_ast(ctx.nCtx, obj) != 0)
01520             {
01521                 switch (sk)
01522                 {
01523                     case Z3_sort_kind.Z3_INT_SORT: return new IntNum(ctx, obj);
01524                     case Z3_sort_kind.Z3_REAL_SORT: return new RatNum(ctx, obj);
01525                     case Z3_sort_kind.Z3_BV_SORT: return new BitVecNum(ctx, obj);
01526                 }
01527             }
01528 
01529             switch (sk)
01530             {
01531                 case Z3_sort_kind.Z3_BOOL_SORT: return new BoolExpr(ctx, obj);
01532                 case Z3_sort_kind.Z3_INT_SORT: return new IntExpr(ctx, obj);
01533                 case Z3_sort_kind.Z3_REAL_SORT: return new RealExpr(ctx, obj);
01534                 case Z3_sort_kind.Z3_BV_SORT: return new BitVecExpr(ctx, obj);
01535                 case Z3_sort_kind.Z3_ARRAY_SORT: return new ArrayExpr(ctx, obj);
01536                 case Z3_sort_kind.Z3_DATATYPE_SORT: return new DatatypeExpr(ctx, obj);
01537             }
01538 
01539             return new Expr(ctx, obj);
01540         }
