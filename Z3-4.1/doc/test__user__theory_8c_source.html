00336 void apply_unit_axiom_for_parents_of(Z3_theory t, Z3_ast n) {
00337     SimpleTheoryData * td = (SimpleTheoryData*)Z3_theory_get_ext_data(t);
00338     Z3_context c = Z3_theory_get_context(t);
00339     /*
00340       The following idiom is the standard approach for traversing
00341       applications of the form
00342       
00343       g(..., n', ...) 
00344       where
00345       - g is an interpreted function symbol of \c t.
00346       - n' is in the same equivalence class of n.
00347     */
00348     Z3_ast n_prime = n;
00349     do {
00350         unsigned num_parents = Z3_theory_get_num_parents(t, n_prime);
00351         unsigned i;
00352         for (i = 0; i < num_parents; i++) {
00353             Z3_app parent = Z3_to_app(c, Z3_theory_get_parent(t, n_prime, i));
00354             /* check whether current parent is of the form f(a, n_prime) */
00355             if (Z3_get_app_decl(c, parent) == td->f && Z3_get_app_arg(c, parent, 1) == n_prime) {
00356                 /* assert f(a, u) = a */
00357                 Z3_ast a = Z3_get_app_arg(c, parent, 0);
00358                 Z3_theory_assert_axiom(t, Z3_mk_eq(c, mk_binary_app(c, td->f, a, td->u), a));
00359                 /* Instead of asserting f(a, u) = a, we could also have asserted
00360                    the clause:
00361                    (not (n_prime = u)) or (f(a, n_prime) = a)
00362                    
00363                    However, this solution is wasteful, because the axiom 
00364                    assert f(a, u) = a is simpler and more general.
00365                    Note that n_prime is in the equivalence class of n,
00366                    and n is now equal to u. So, n_prime is also equal to u.
00367                    Then, using congruence, Z3 will also deduce that f(a, n_prime) = a
00368                    using the simpler axiom.
00369                 */
00370             }
00371             /* check whether current parent is of the form f(n_prime, a) */
00372             if (Z3_get_app_decl(c, parent) == td->f && Z3_get_app_arg(c, parent, 0) == n_prime) {
00373                 /* assert f(u, a) = a */
00374                 Z3_ast a = Z3_get_app_arg(c, parent, 1);
00375                 Z3_theory_assert_axiom(t, Z3_mk_eq(c, mk_binary_app(c, td->f, td->u, a), a));
00376             }
00377         }
00378         n_prime = Z3_theory_get_eqc_next(t, n_prime);
00379     }
00380     while (n_prime != n);
00381 }

00704 void apply_compare_axiom_for_parents_of(Z3_theory t, Z3_ast n, Z3_ast v) {
00705     PATheoryData * td = (PATheoryData*)Z3_theory_get_ext_data(t);
00706     Z3_context c = Z3_theory_get_context(t);
00707     Z3_ast n_prime = Z3_theory_get_eqc_root(t, n);
00708     do {
00709         unsigned num_parents = Z3_theory_get_num_parents(t, n_prime);
00710         unsigned i;
00711         for (i = 0; i < num_parents; i++) {
00712             Z3_app parent = Z3_to_app(c, Z3_theory_get_parent(t, n_prime, i));
00713             if (Z3_get_app_decl(c, parent) == td->Compare) {
00714                 Z3_ast arg1 = Z3_get_app_arg(c, parent, 0);
00715                 Z3_ast arg2 = Z3_get_app_arg(c, parent, 1);
00716                 if (Z3_theory_get_eqc_root(t, arg1) == n)
00717                     arg1 = v;
00718                 else
00719                     arg1 = get_eqc_value(t, arg1);
00720                 if (Z3_theory_get_eqc_root(t, arg2) == n)
00721                     arg2 = v;
00722                 else
00723                     arg2 = get_eqc_value(t, arg2);
00724                 if (arg1 != 0 && arg2 != 0) {
00725                     switch (Compare(t, arg1, arg2)) {
00726                     case Z3_L_TRUE:
00727                         // assert axiom: Compare(arg1, arg2)
00728                         Z3_theory_assert_axiom(t, mk_binary_app(c, td->Compare, arg1, arg2));
00729                         break;
00730                     case Z3_L_FALSE:
00731                         // assert axiom: !Compare(arg1, arg2)
00732                         Z3_theory_assert_axiom(t, Z3_mk_not(c, mk_binary_app(c, td->Compare, arg1, arg2)));
00733                         break;
00734                     case Z3_L_UNDEF:
00735                         // do nothing
00736                         break; 
00737                     }
00738                 }
00739             }
00740         }
00741         n_prime = Z3_theory_get_eqc_next(t, n_prime);
00742     }
00743     while (n_prime != n);

00766 Z3_bool PATh_final_check(Z3_theory t) {
00767     PATheoryData * td = (PATheoryData*)Z3_theory_get_ext_data(t);
00768     Z3_context c = Z3_theory_get_context(t);
00769     unsigned i, num;
00770     printf("Final check\n");
00771     /* check whether all (relevant) Compare(n1, n2) applications could be evaluated */
00772     num = Z3_theory_get_num_apps(t);
00773     for (i = 0; i < num; i++) {
00774         Z3_ast curr    = Z3_theory_get_app(t, i);
00775         Z3_func_decl d = Z3_get_app_decl(c, Z3_to_app(c, curr));
00776         if (d == td->Compare) {
00777             Z3_ast arg1 = Z3_get_app_arg(c, Z3_to_app(c, curr), 0);
00778             Z3_ast arg2 = Z3_get_app_arg(c, Z3_to_app(c, curr), 1);
00779             if (get_eqc_value(t, arg1) == 0 || get_eqc_value(t, arg2) == 0) {
00780                 printf("failed to evaluate Compare(%s", Z3_ast_to_string(c, arg1));
00781                 printf(", %s)\n", Z3_ast_to_string(c, arg2));
00782                 return Z3_FALSE; /* giving up... could not evaluate this Compare application */
00783             }
00784         }
00785     }
00786     return Z3_TRUE;

00792 Z3_theory mk_pa_theory(Z3_context ctx) {
00793     Z3_sort compare_domain[2];
00794     Z3_symbol string_name  = Z3_mk_string_symbol(ctx, "String");
00795     Z3_symbol compare_name = Z3_mk_string_symbol(ctx, "Compare");
00796     Z3_sort B              = Z3_mk_bool_sort(ctx);
00797     PATheoryData * td      = (PATheoryData *)malloc(sizeof(PATheoryData));
00798     Z3_theory Th           = Z3_mk_theory(ctx, "CompareProcedurealAttachment", td);
00799     td->String             = Z3_theory_mk_sort(ctx, Th, string_name); 
00800     compare_domain[0] = td->String; 
00801     compare_domain[1] = td->String;
00802     td->Compare            = Z3_theory_mk_func_decl(ctx, Th, compare_name, 2, compare_domain, B);
00803 
00804     Z3_set_delete_callback(Th, PATh_delete);
00805     Z3_set_reduce_app_callback(Th, PATh_reduce_app);
00806     Z3_set_new_eq_callback(Th, PATh_new_eq);
00807     Z3_set_final_check_callback(Th, PATh_final_check);
00808     return Th;
00809 }
