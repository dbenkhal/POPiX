00799     template<typename T>
00800     struct coerce {
00801         virtual ~coerce() {}
00802         virtual T* operator()(ast* a) const = 0; 
00803     };
00804 
00805     struct sort_coerce : public coerce<sort> {
00806         virtual sort* operator()(ast* a) const { 
00807             if (!a || a->get_kind() != AST_SORT) {
00808                 return 0;
00809             }
00810             return to_sort(a); 
00811         }
00812     };
00813 
00814     struct func_decl_coerce : public coerce<func_decl> {
00815         virtual func_decl* operator()(ast* a) const { 
00816             if (!a || a->get_kind() != AST_FUNC_DECL) {
00817                 return 0;
00818             }
00819             return to_func_decl(a); 
00820         }
00821     };
00822 
00823     struct pattern_coerce : public coerce<app> {
00824         ast_manager& m_manager;
00825         pattern_coerce(ast_manager& m): m_manager(m) {}
00826         virtual app* operator()(ast* a) const { 
00827             if (!a || a->get_kind() != AST_APP) {
00828                 return 0;
00829             }
00830             if (!m_manager.is_pattern(to_app(a))) {
00831                 return 0;
00832             }
00833             return to_app(a);
00834         }
00835     };
00836 
00837     struct expr_coerce : public coerce<expr> {
00838         virtual expr* operator()(ast* a) const { 
00839             if (!a) {
00840                 return 0;
00841             }
00842             ast_kind k = a->get_kind();
00843             switch(k) {
00844             case AST_APP:
00845             case AST_QUANTIFIER:
00846             case AST_VAR:
00847                 return reinterpret_cast<expr*>(a);
00848             default:
00849                 return 0;
00850             }
00851         }
00852     };
00853 
00854     struct app_coerce : public coerce<app> {
00855         virtual app* operator()(ast* a) const { 
00856             if (!a) {
00857                 return 0;
00858             }
00859             if (a->get_kind() == AST_APP) {
00860                 return to_app(a);
00861             }
00862             return 0;
00863         }
00864     };

00867     template<typename T>
00868     bool parse_ast(symbol id, T*& n, const coerce<T>& coerce)
00869     {
00870         ast* a;                                                         
00871         if (!m_table.find(id, a)) {
00872             warning_msg("line %d. Could not find id '%s'\n", 
00873                         m_line, id.str().c_str());    
00874             return false;                                               
00875         }                              
00876         n = coerce(a);
00877         if (n == 0) {
00878             warning_msg("line %d. Wrong kind %d for %s\n", 
00879                         m_line, a->get_kind(), id.str().c_str());             
00880             return false;                                               
00881         }                                                               
00882         return true;                                                       
00883     }

00886     template<typename T>
00887     bool parse_ast(T*& n, const coerce<T>& coerce)
00888     {
00889         symbol id;                                                      
00890         if (parse_id(id)) { 
00891             return parse_ast(id, n, coerce);
00892         }                                                                   
00893         return false;                                                       
00894     }

00897     template<typename T>
00898     bool parse_asts(ref_vector<T, ast_manager>& asts, const coerce<T>& c)
00899     {
00900         symbol id;
00901         T*   n;
00902         while (m_string[0] != '\n') {
00903             if (strcmp(m_string.c_str(), "BUILTIN") == 0) {
00904                 return true;
00905             }
00906             if (!parse_ast(n,c)) {
00907                 return false;
00908             }
00909             asts.push_back(n);
00910         }
00911         return true;
00912     }
